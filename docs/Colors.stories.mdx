import {Canvas, Meta, Story} from '@storybook/addon-docs/blocks'
import DocTable from './DocTable'

import tailwindConfig from '../tailwind.config.js'

export const ColorTableHead = () => {
  return (
    <thead>
      <tr>
        <th>Color</th>
        <th>Class</th>
        <th>Value</th>
      </tr>
    </thead>
  )
}

export const ColorTableRow = ({colorClassName, value}) => {
  return (
    <tr key={colorClassName}>
      <td>
        <div
          className={`w-80 h-40 rounded border-solid border bg-${colorClassName}`}
          style={{borderColor: '#d1d1d1'}}
        />
      </td>
      <td>
        <code>{colorClassName}</code>
      </td>
      <td>
        <code>{value}</code>
      </td>
    </tr>
  )
}

<Meta title="Docs/Colors" />

# Primary

Primary colors.

<DocTable>
  <>
    <ColorTableHead />
    <tbody>
      {Object.entries(tailwindConfig.theme.colors.primary).map((color, i) => {
        return (
          <ColorTableRow
            key={color[1]}
            colorClassName={
              color[0].endsWith('DEFAULT') ? 'primary' : `primary-${color[0]}`
            }
            value={color[1]}
          />
        )
      })}
    </tbody>
  </>
</DocTable>
