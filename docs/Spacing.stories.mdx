export const SpacingTableRow = ({spacingClassName, size}) => {
  return (
    <tr key={spacingClassName}>
      <td>
        <div className={`h-20 w-${spacingClassName} bg-primary`}></div>
      </td>
      <td>
        <code>{spacingClassName}</code>
      </td>
      <td>
        <code>{size}</code>
      </td>
      <td>
        <code>
          {spacingClassName === 'px' ? '1px' : `${parseFloat(size) * 16}px`}
        </code>
      </td>
    </tr>
  )
}

# Spacing

The spacing scale is defined in rem which is set to `16px` at the time this page was created.

Values in the scale are listed below, along with each pixel equivalent.

**Note -**
If the rem value is changed, the calculation on this page will be wrong unless updated.
